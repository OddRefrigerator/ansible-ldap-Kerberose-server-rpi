KRB5 service:
----------------------------
sudo ./setupHost.sh
sudo apt install krb5-{kdc,admin-server,config,user,kdc-ldap} libsasl2-{dev,modules-gssapi-mit,modules} libpam-krb5 sssd-krb5 -y
sudo systemctl stop krb5-kdc
sudo systemctl stop krb5-admin-server
sudo kdb5_util create -s -r HOME.COM -P password123
sudo cp kadm5.acl /etc/krb5kdc/kadm5.acl
sudo cp krb5.conf /etc/krb5.conf
sudo cp kdc.conf /etc/krb5kdc/kdc.conf
sudo systemctl start krb5-kdc
sudo systemctl start krb5-admin-server

LDAP service
----------------------------
sudo apt install -y slapd ldap-utils schema2ldif libnss-ldapd
sudo cp ldap.conf  /etc/ldap/ldap.conf 
sudo cp slapd.conf /usr/share/slapd/slapd.conf
sudo ./populateldap.sh
sudo systemctl restart krb5-kdc
sudo systemctl restart krb5-admin-server
sudo systemctl restart slapd
sudo ./installldapscripts.sh
